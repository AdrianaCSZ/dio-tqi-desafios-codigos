        Scanner scan = new Scanner(System.in);

        int N = scan.nextInt();

        // quebra string em várias substrings a partir de um caracter
        String[] s = string.split(" ");

        double[] idade = new double[N];

        for (int i = 0; i < N; i++)

        {
